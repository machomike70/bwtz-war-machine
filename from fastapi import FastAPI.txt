from fastapi import FastAPI
from pydantic import BaseModel
from typing import Optional, List

app = FastAPI(title="XtremeRippleProtocol X Growth Bot")

@app.get("/")
def root():
    return {
        "status": "ok",
        "service": "agent-orchestrator",
        "company": "XtremeRippleProtocol LLC",
        "white_label_service": True,
        "pricing": {
            "Free": "Limited to 5 posts/day",
            "Basic": "6 XRP per month - Unlimited posts & replies",
            "Pro": "12 XRP per month - Full features + daily batch generation"
        }
    }

class OrchestrateRequest(BaseModel):
    task: str
    account: Optional[str] = None
    context: Optional[str] = None
    client_brand: Optional[str] = None
    plan: Optional[str] = "Free"

@app.post("/orchestrate")
def orchestrate(request: OrchestrateRequest):
    task_lower = request.task.lower()
    acc = request.account or "@Mmozley70"
    brand = request.client_brand or "XtremeRippleProtocol LLC"
    plan = request.plan or "Free"

    # Daily batch generation - Pro only
    if any(word in task_lower for word in ["daily", "batch", "5 posts"]):
        if plan == "Free":
            return {
                "status": "limited",
                "message": "Daily batch generation is a Pro feature (12 XRP/month). Upgrade for 5 ready posts per day.",
                "powered_by": brand
            }
        posts = [
            f"🔥 {request.context or 'XtremeRippleProtocol'} just hit new levels. Ape in and hold on we about to take a ride fr fr 🚀 Who's coming with on @{acc.strip('@')}? #XRPL #TextRp",
            f"GM frens 🔥 @{acc.strip('@')} checking in. XtremeRippleProtocol vibes strong today. Let’s go peeps! #TextRp",
            f"Quick update: {request.context or 'XtremeRippleProtocol'} community growing fast. Thanks for pulling up. Ape in! 🚀",
            f"Love the energy in the replies 🔥 Keep building with us. Who's ready for the next leg? #XRPL",
            f"Delegation reminder for @textrpsms holders: Simple, fast, and rewarding. Join the movement! #TextRp"
        ]
        return {
            "status": "success",
            "account": acc,
            "daily_posts": posts,
            "plan_used": plan,
            "powered_by": brand,
            "tip": "Post one in the morning, one in the evening. Rotate across your 5 accounts."
        }

    # Single post generation
    if any(word in task_lower for word in ["post", "tweet", "generate post"]):
        post = f"🔥 {request.context or 'XtremeRippleProtocol'} vibes are crazy right now. Ape in and hold on we about to take a ride fr fr 🚀 Who's coming with on @{acc.strip('@')}? #XRPL #TextRp"
        return {
            "status": "success",
            "account": acc,
            "generated_post": post,
            "powered_by": brand,
            "plan_used": plan
        }

    elif any(word in task_lower for word in ["reply", "response", "answer"]):
        return {
            "status": "success",
            "account": acc,
            "suggested_reply": "Love that bro 🔥 Thanks for pulling up. Let’s go peeps!",
            "powered_by": brand,
            "plan_used": plan
        }

    else:
        return {
            "status": "success",
            "powered_by": brand,
            "plan": plan,
            "message": "White-label X bot ready. Try task='daily posts' on Pro plan for 5 posts at once."
        }
